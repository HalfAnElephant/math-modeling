# 重建搜索全枚举验证 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` or `superpowers:executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 对 16 个目标的直接确认集合做全枚举验证，说明重建搜索方案在直接确认集合选择层面是否达到最优或近最优。

**Architecture:** 新增 `c_uav_inspection/exact.py`。它遍历 `2^16=65536` 个直接确认集合，调用现有 `_rebuild_for_direct_set` 评估可行性和闭环指标，再按闭环时间和归一化综合目标排名，与 `solve_joint_problem_for_k` 的重建搜索结果比较。

**Tech Stack:** Python 3.11, standard library `dataclasses`, existing `problem2`, `objective`, `model`, `experiments`, `pytest`, `json`, `csv`.

```python
def _evaluate_direct_set(
    data: ProblemData,
    k: int,
    direct_nodes: tuple[int, ...],
    direct_threshold_multiplier: float,
    allow_split_hover: bool,
) -> DirectSetEvaluation:
    sol = _rebuild_for_direct_set(
        data=data,
        k=k,
        direct_nodes=direct_nodes,
        direct_threshold_multiplier=direct_threshold_multiplier,
        allow_split_hover=allow_split_hover,
    )
    if sol is None:
        return DirectSetEvaluation(
            direct_nodes=direct_nodes,
            feasible=False,
            closed_loop_time_s=None,
            uav_phase_time_s=None,
            ground_review_time_s=None,
            manual_count=None,
            weighted_manual_cost=None,
            direct_confirm_count=len(direct_nodes),
            total_energy_j=None,
            load_std_s=None,
            route_count=None,
            normalized_objective=None,
        )

    summary = summarize_uav_solution(sol.routes, data, data.params.battery_swap_time_s)
    return DirectSetEvaluation(
        direct_nodes=tuple(sol.closed_loop.direct_confirmed_nodes),
        feasible=True,
        closed_loop_time_s=sol.closed_loop.closed_loop_time_s,
        uav_phase_time_s=sol.closed_loop.uav_phase_time_s,
        ground_review_time_s=sol.closed_loop.ground_review_time_s,
        manual_count=sol.closed_loop.manual_count,
        weighted_manual_cost=sol.closed_loop.weighted_manual_cost,
        direct_confirm_count=len(sol.closed_loop.direct_confirmed_nodes),
        total_energy_j=summary.total_energy_j,
        load_std_s=summary.load_std_s,
        route_count=len(sol.routes),
        normalized_objective=None,
    )
```

## 4. 归一化目标计算

- [ ] **Step 5: 实现 `_with_normalized_objectives`**

```python
ENUMERATION_WEIGHTS: dict[str, float] = {
    "closed_loop_time_s": 0.45,
    "ground_review_time_s": 0.20,
    "weighted_manual_cost": 0.15,
    "manual_count": 0.10,
    "total_energy_j": 0.05,
    "load_std_s": 0.05,
}


def _with_normalized_objectives(
    evaluations: list[DirectSetEvaluation],
) -> list[DirectSetEvaluation]:
    feasible = [ev for ev in evaluations if ev.feasible]
    rows: list[dict[str, float]] = []
    for ev in feasible:
        rows.append({
            "closed_loop_time_s": float(ev.closed_loop_time_s),
            "ground_review_time_s": float(ev.ground_review_time_s),
            "weighted_manual_cost": float(ev.weighted_manual_cost),
            "manual_count": float(ev.manual_count),
            "total_energy_j": float(ev.total_energy_j),
            "load_std_s": float(ev.load_std_s),
        })

    bounds = bounds_from_candidates(rows, list(ENUMERATION_WEIGHTS))
    scored: list[DirectSetEvaluation] = []
    feasible_iter = iter(feasible)
    score_by_direct_nodes: dict[tuple[int, ...], float] = {}
    for ev, row in zip(feasible, rows):
        score_by_direct_nodes[ev.direct_nodes] = weighted_normalized_objective(
            row,
            bounds,
            ENUMERATION_WEIGHTS,
        )

    for ev in evaluations:
        if ev.feasible:
            scored.append(replace(ev, normalized_objective=score_by_direct_nodes[ev.direct_nodes]))
        else:
            scored.append(ev)
    return scored
```

删除未使用变量 `feasible_iter`，如果实现时使用了它则不要留下。

## 5. 全枚举主函数

- [ ] **Step 6: 实现 `enumerate_direct_confirm_sets`**

```python
def enumerate_direct_confirm_sets(
    data: ProblemData,
    k: int,
    direct_threshold_multiplier: float,
    allow_split_hover: bool = True,
    top_n: int = 20,
) -> DirectSetEnumerationResult:
    target_ids = tuple(target.node_id for target in data.targets)
    evaluations: list[DirectSetEvaluation] = []

    for mask in range(1 << len(target_ids)):
        direct_nodes = tuple(
            target_ids[idx]
            for idx in range(len(target_ids))
            if mask & (1 << idx)
        )
        evaluations.append(_evaluate_direct_set(
            data=data,
            k=k,
            direct_nodes=direct_nodes,
            direct_threshold_multiplier=direct_threshold_multiplier,
            allow_split_hover=allow_split_hover,
        ))

    evaluations = _with_normalized_objectives(evaluations)
    feasible = [ev for ev in evaluations if ev.feasible]
    if not feasible:
        raise RuntimeError("No feasible direct-confirm set found during enumeration")
```

```python
    rebuild = solve_joint_problem_for_k(
        data,
        k=k,
        direct_threshold_multiplier=direct_threshold_multiplier,
        allow_split_hover=allow_split_hover,
    )
    rebuild_direct = tuple(rebuild.closed_loop.direct_confirmed_nodes)
    rebuild_eval = next(ev for ev in feasible if ev.direct_nodes == rebuild_direct)

    rebuild_time_rank = next(idx for idx, ev in enumerate(by_time, start=1) if ev.direct_nodes == rebuild_direct)
    rebuild_objective_rank = next(idx for idx, ev in enumerate(by_objective, start=1) if ev.direct_nodes == rebuild_direct)

    best_time = by_time[0]
    best_obj = by_objective[0]
    gap_s = float(rebuild_eval.closed_loop_time_s) - float(best_time.closed_loop_time_s)
    gap_pct = gap_s / float(best_time.closed_loop_time_s) * 100.0
    obj_gap = float(rebuild_eval.normalized_objective) - float(best_obj.normalized_objective)

    return DirectSetEnumerationResult(
        total_subsets=1 << len(target_ids),
        feasible_subsets=len(feasible),
        best_by_closed_loop=best_time,
        best_by_objective=best_obj,
        rebuild_solution=rebuild_eval,
        rebuild_time_rank=rebuild_time_rank,
        rebuild_time_gap_s=gap_s,
        rebuild_time_gap_pct=gap_pct,
        rebuild_objective_rank=rebuild_objective_rank,
        rebuild_objective_gap=obj_gap,
        top_by_closed_loop=tuple(by_time[:top_n]),
        top_by_objective=tuple(by_objective[:top_n]),
    )
```

```markdown
由于本题目标数为16，直接确认集合共有 $2^{16}=65536$ 种。本文对所有直接确认集合进行枚举：对每个集合重新构造无人机路线、求解物业人员 TSP，并计算闭环时间和归一化综合目标。该枚举只验证直接确认集合选择层面的最优性，底层航次构造仍使用本文的启发式路线生成器，因此不能表述为完整 VRP 全局最优。
```

结论规则：

- `rebuild_time_gap_pct <= 1.0` 且 `rebuild_objective_rank` 靠前：可写“近最优”。
- `rebuild_time_gap_pct == 0` 且 `rebuild_objective_gap == 0`：可写“达到枚举最优”。
- 差距明显时，推荐方案必须改为枚举最优，不可继续推荐重建搜索旧结果。
