def split_order_into_energy_feasible_routes_no_split(
    order: tuple[int, ...],
    hover_times_s: Mapping[int, float],
    data: ProblemData,
) -> tuple[UAVRoute, ...]:
    """Split a visitation order into sorties without splitting any target hover demand.

    Each positive-demand target must be fully served in exactly one sortie.
    """

- [ ] **Step 2: 复制必要输入校验**

函数开头必须包含与可拆分版本一致的关键校验：

```python
negative_nodes = [(nid, val) for nid, val in hover_times_s.items() if val < 0.0]
if negative_nodes:
    raise ValueError(
        f"hover_times_s values must be non-negative. Negative values detected: {negative_nodes}"
    )

known = _known_node_ids(data)
unknown_order = sorted(nid for nid in order if nid not in known)
unknown_hover = sorted(nid for nid in hover_times_s if nid not in known)
if unknown_order:
    raise ValueError(f"order contains node IDs that do not exist in the flight matrix: {unknown_order}")
if unknown_hover:
    raise ValueError(f"hover_times_s contains node IDs that do not exist in the flight matrix: {unknown_hover}")

order_set = set(order)
for node_id, demand_s in hover_times_s.items():
    if demand_s > EPSILON and node_id not in order_set:
        raise ValueError(
            f"Node {node_id} has positive hover demand but is not present in the visitation order"
        )

limit_j = data.params.effective_energy_limit_j
hover_power = data.params.hover_power_j_per_s
if hover_power <= 0.0:
    raise ValueError(f"hover_power_j_per_s must be positive, got {hover_power}")
```

- [ ] **Step 3: 校验单目标完整服务可行**

```python
for node_id, demand_s in hover_times_s.items():
    if demand_s <= EPSILON:
        continue
    single_energy = (
        data.flight_energy_j[(0, node_id)]
        + demand_s * hover_power
        + data.flight_energy_j[(node_id, 0)]
    )
    if single_energy > limit_j + EPSILON:
        raise InfeasibleError(
            f"Target {node_id} requires {single_energy:.2f} J for unsplit service, "
            f"exceeding single-sortie limit {limit_j:.2f} J"
        )
```

- [ ] **Step 4: 实现不可拆分装箱主循环**

```python
routes: list[UAVRoute] = []
sortie_id = 0
node_sequence: list[int] = [0]
route_hover: dict[int, float] = {}
current_node = 0
energy_used_without_return = 0.0

def close_route() -> None:
    nonlocal sortie_id, node_sequence, route_hover, current_node, energy_used_without_return
    if len(node_sequence) == 1:
        return
    node_sequence.append(0)
    sortie_id += 1
    routes.append(UAVRoute(
        uav_id=0,
        sortie_id=sortie_id,
        node_sequence=tuple(node_sequence),
        hover_times_s=dict(route_hover),
    ))
    node_sequence = [0]
    route_hover = {}
    current_node = 0
    energy_used_without_return = 0.0

for node_id in order:
    demand_s = hover_times_s.get(node_id, 0.0)
    if demand_s <= EPSILON:
        continue

    arrival_energy = data.flight_energy_j[(current_node, node_id)]
    return_energy = data.flight_energy_j[(node_id, 0)]
    full_service_energy_if_added = (
        energy_used_without_return
        + arrival_energy
        + demand_s * hover_power
        + return_energy
    )

    if full_service_energy_if_added > limit_j + EPSILON:
        close_route()
        arrival_energy = data.flight_energy_j[(0, node_id)]
        return_energy = data.flight_energy_j[(node_id, 0)]
        full_service_energy_if_added = arrival_energy + demand_s * hover_power + return_energy
        if full_service_energy_if_added > limit_j + EPSILON:
            raise InfeasibleError(
                f"Target {node_id} cannot be served unsplit within one sortie"
            )

    node_sequence.append(node_id)
    route_hover[node_id] = demand_s
    energy_used_without_return += arrival_energy + demand_s * hover_power
    current_node = node_id

close_route()
return tuple(routes)
```

def _run_problem2_split_hover_ablation(data: ProblemData, output_dir: Path) -> None:
    rows: list[dict[str, Any]] = []
    k = data.params.k_max
    multiplier = 1.0
    sw = data.params.battery_swap_time_s

    for scheme, allow_split in (
        ("可拆分悬停", True),
        ("不可拆分悬停", False),
    ):
        try:
            sol = solve_joint_problem_for_k(
                data,
                k,
                multiplier,
                allow_split_hover=allow_split,
            )
            summary = summarize_uav_solution(sol.routes, data, sw)
            rows.append({
                "scheme": scheme,
                "allow_split_hover": allow_split,
                "k": k,
                "direct_threshold_multiplier": multiplier,
                "closed_loop_time_s": sol.closed_loop.closed_loop_time_s,
                "uav_phase_time_s": sol.closed_loop.uav_phase_time_s,
                "ground_review_time_s": sol.closed_loop.ground_review_time_s,
                "manual_count": sol.closed_loop.manual_count,
                "weighted_manual_cost": sol.closed_loop.weighted_manual_cost,
                "direct_confirm_count": len(sol.closed_loop.direct_confirmed_nodes),
                "total_energy_j": summary.total_energy_j,
                "load_std_s": summary.load_std_s,
                "route_count": len(sol.routes),
                "feasible": True,
            })
        except InfeasibleError:
            rows.append({
                "scheme": scheme,
                "allow_split_hover": allow_split,
                "k": k,
                "direct_threshold_multiplier": multiplier,
                "closed_loop_time_s": "",
                "uav_phase_time_s": "",
                "ground_review_time_s": "",
                "manual_count": "",
                "weighted_manual_cost": "",
                "direct_confirm_count": "",
                "total_energy_j": "",
                "load_std_s": "",
                "route_count": "",
                "feasible": False,
            })

    _write_csv(output_dir / "problem2_split_hover_ablation.csv", rows)
```

def test_no_split_routes_never_split_positive_hover():
    data = load_problem_data(DATA_PATH)
    order = (1, 2, 3)
    hover = {1: 50.0, 2: 60.0, 3: 35.0}

    routes = split_order_into_energy_feasible_routes_no_split(order, hover, data)

    served_by_node = {}
    route_count_by_node = {}
    for route in routes:
        for node_id, seconds in route.hover_times_s.items():
            served_by_node[node_id] = served_by_node.get(node_id, 0.0) + seconds
            route_count_by_node[node_id] = route_count_by_node.get(node_id, 0) + 1

    assert served_by_node == pytest.approx(hover)
    assert all(count == 1 for count in route_count_by_node.values())
    assert all(evaluate_uav_route(route, data).feasible_energy for route in routes)
```
