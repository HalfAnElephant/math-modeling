```python
def _run_problem2_acceptance_tolerance_sensitivity(data: ProblemData, output_dir: Path) -> None:
    rows: list[dict[str, Any]] = []
    k = data.params.k_max
    multiplier = 1.0
    sw = data.params.battery_swap_time_s

    for tol in (1.00, 1.03, 1.05, 1.10):
        sol = solve_joint_problem_for_k(
            data,
            k,
            multiplier,
            manual_reduction_time_tolerance=tol,
        )
        summary = summarize_uav_solution(sol.routes, data, sw)
        rows.append({
            "manual_reduction_time_tolerance": tol,
            "closed_loop_time_s": sol.closed_loop.closed_loop_time_s,
            "uav_phase_time_s": sol.closed_loop.uav_phase_time_s,
            "ground_review_time_s": sol.closed_loop.ground_review_time_s,
            "manual_count": sol.closed_loop.manual_count,
            "weighted_manual_cost": sol.closed_loop.weighted_manual_cost,
            "direct_confirm_count": len(sol.closed_loop.direct_confirmed_nodes),
            "total_energy_j": summary.total_energy_j,
            "load_std_s": summary.load_std_s,
            "direct_confirmed_nodes": _node_tuple_to_string(sol.closed_loop.direct_confirmed_nodes),
            "manual_target_nodes": _node_tuple_to_string(sol.closed_loop.manual_target_nodes),
        })

    _write_csv(output_dir / "problem2_acceptance_tolerance_sensitivity.csv", rows)
```

```python
def _run_problem2_energy_limit_sensitivity(data: ProblemData, output_dir: Path) -> None:
    rows: list[dict[str, Any]] = []
    base_limit = data.params.effective_energy_limit_j

    for limit_j in (0.90 * base_limit, base_limit, 1.10 * base_limit):
        scenario = replace(
            data,
            params=replace(data.params, effective_energy_limit_j=limit_j),
        )
        try:
            sol = solve_joint_problem_for_k(scenario, scenario.params.k_max, 1.0)
            summary = summarize_uav_solution(sol.routes, scenario, scenario.params.battery_swap_time_s)
            rows.append({
                "effective_energy_limit_j": limit_j,
                "closed_loop_time_s": sol.closed_loop.closed_loop_time_s,
                "uav_phase_time_s": sol.closed_loop.uav_phase_time_s,
                "ground_review_time_s": sol.closed_loop.ground_review_time_s,
                "manual_count": sol.closed_loop.manual_count,
                "weighted_manual_cost": sol.closed_loop.weighted_manual_cost,
                "direct_confirm_count": len(sol.closed_loop.direct_confirmed_nodes),
                "total_energy_j": summary.total_energy_j,
                "load_std_s": summary.load_std_s,
                "route_count": len(sol.routes),
                "feasible": True,
            })
        except InfeasibleError:
            rows.append({
                "effective_energy_limit_j": limit_j,
                "closed_loop_time_s": "",
                "uav_phase_time_s": "",
                "ground_review_time_s": "",
                "manual_count": "",
                "weighted_manual_cost": "",
                "direct_confirm_count": "",
                "total_energy_j": "",
                "load_std_s": "",
                "route_count": "",
                "feasible": False,
            })

    _write_csv(output_dir / "problem2_energy_limit_sensitivity.csv", rows)
```

## 7. 论文同步

- [ ] **Step 13: 新增接受准则敏感性小节**

位置：第 6 章问题2敏感性分析之后。

标题：

```markdown
## 6.X 重建搜索接受准则敏感性
```

表格：

```markdown
表X 接受准则容忍倍数敏感性

| $\rho$ | $T$/s | $T_u$/s | $T_g$/s | 人工点数 | $C_M$ | 直接确认数 |
|---:|---:|---:|---:|---:|---:|---:|
```

正文解释：

```markdown
其中 $\rho=1.00$ 表示只有在闭环时间不变差时才接受人工负担下降；$\rho=1.03$ 为本文推荐设置；$\rho=1.05$ 和 $1.10$ 表示更愿意以少量闭环时间代价换取人工负担降低。人工负担同时包括加权人工代价 $C_M$ 和人工点数 $|M|$。
```
