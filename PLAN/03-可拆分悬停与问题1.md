from pathlib import Path

import pytest

from c_uav_inspection.data import load_problem_data
from c_uav_inspection.model import evaluate_uav_route
from c_uav_inspection.search import nearest_neighbor_order, split_order_into_energy_feasible_routes


DATA_PATH = Path("2026同济数学建模竞赛赛题/2026C数据.xlsx")


def test_nearest_neighbor_order_visits_each_target_once():
    data = load_problem_data(DATA_PATH)
    order = nearest_neighbor_order(data, [target.node_id for target in data.targets])

    assert sorted(order) == list(range(1, 17))
    assert len(order) == 16


def test_split_order_into_energy_feasible_routes_satisfies_base_hover():
    data = load_problem_data(DATA_PATH)
    order = nearest_neighbor_order(data, [target.node_id for target in data.targets])
    hover = {target.node_id: target.base_hover_time_s for target in data.targets}

    routes = split_order_into_energy_feasible_routes(order, hover, data)
    served = {node_id: 0.0 for node_id in hover}
    for route in routes:
        for node_id, seconds in route.hover_times_s.items():
            served[node_id] += seconds

    assert len(routes) >= 2
    assert all(route.node_sequence[0] == 0 and route.node_sequence[-1] == 0 for route in routes)
    assert all(evaluate_uav_route(route, data).feasible_energy for route in routes)
    assert served == pytest.approx(hover)


def test_split_order_allows_one_target_hover_to_span_multiple_sorties():
    data = load_problem_data(DATA_PATH)
    target_id = 16
    single_visit_hover_capacity = (
        data.params.effective_energy_limit_j
        - data.flight_energy_j[(0, target_id)]
        - data.flight_energy_j[(target_id, 0)]
    ) / data.params.hover_power_j_per_s
    hover = {target_id: single_visit_hover_capacity + 30.0}

    routes = split_order_into_energy_feasible_routes((target_id,), hover, data)
    served = sum(route.hover_times_s.get(target_id, 0.0) for route in routes)

    assert len(routes) >= 2
    assert served == pytest.approx(hover[target_id])
    assert all(evaluate_uav_route(route, data).feasible_energy for route in routes)

## 2. 实现搜索工具

创建 `c_uav_inspection/search.py`，必须实现：

- `EPSILON = 1e-7`
- `nearest_neighbor_order(data, node_ids)`
  - 从节点 0 出发。
  - 每次选飞行时间最短的剩余节点。
  - 若飞行时间相同，选编号更小的节点。

- `split_order_into_energy_feasible_routes(order, hover_times_s, data)`
  - 输入 `hover_times_s` 是每个目标的总悬停需求。
  - 每一趟路线都从 0 出发并回到 0。
  - 若当前趟无法完整服务某目标，则按以下公式服务部分悬停：

```text
served_hover_s = min(remaining_hover_s, residual_energy_J / hover_power_j_per_s)
```

  - 服务部分悬停后结束当前趟，剩余悬停需求进入下一趟。
  - 如果某目标 0 往返且悬停 0 秒都超过单趟能耗上限，抛出 `ValueError`。
  - 返回值是 `tuple[UAVRoute, ...]`。

## 4. 实现问题 1 求解器

创建 `c_uav_inspection/problem1.py`，必须包含：

- `Problem1Solution`
  - `routes`
  - `total_hover_by_node`
  - `summary`

- `_assign_routes_to_uavs(routes, k, data, battery_swap_time_s)`
  - 初始化 `work_times = {1: 0.0, ..., k: 0.0}`。
  - 按每趟悬停时间从大到小分配。
  - 每次分给当前累计工作时长最短的无人机。
  - 每架无人机第二趟及以后计入一次换电时间。
  - 返回按 `(uav_id, sortie_id)` 排序的路线元组。

- `solve_uav_hover_plan(data, k, battery_swap_time_s, hover_requirements_s, improve=False)`
  - 这是通用无人机悬停需求求解器。
  - 先用 `nearest_neighbor_order` 生成顺序。
  - 再用 `split_order_into_energy_feasible_routes` 生成能耗可行趟次。
  - `improve=True` 时对每趟使用 2-opt。
  - 汇总每个目标累计悬停。

- `solve_problem1_for_k(data, k, battery_swap_time_s, improve=False)`
  - 构造 `{target.node_id: target.base_hover_time_s}`。
  - 调用 `solve_uav_hover_plan`。

同时在 `c_uav_inspection/search.py` 追加：

- `improve_route_by_two_opt(route, data)`
  - 固定一趟包含的目标集合和悬停时间。
  - 只反转访问顺序。
  - 只接受能耗可行且路线时长更短的候选。
